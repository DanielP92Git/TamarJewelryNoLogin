    // D-07/D-08: change a cart line's quantity (the `amount` field). Min 1; capped
    // at the line's stock (`quantity`). Persists via the existing createLocalStorage
    // for guests; for logged-in users mirrors the existing add path so the server count
    // stays in sync on increase.
    export const increaseAmount = async function (itemId) {
      const item = cart.find(el => el.id == itemId);          // == : id may be String/Number
      if (!item) return;
      const stock = Number(item.quantity) || Infinity;        // quantity is the stock cap
      if ((Number(item.amount) || 1) >= stock) return item.amount; // at cap, no-op
      item.amount = (Number(item.amount) || 1) + 1;
      if (!localStorage.getItem('auth-token')) createLocalStorage();
      else { try { await fetch(`${host}/addtocart`, { method:'POST', headers:{ Accept:'application/form-data','auth-token':`${localStorage.getItem('auth-token')}`,'Content-Type':'application/json' }, body: JSON.stringify({ itemId }) }); } catch (e) { console.error('increaseAmount sync failed', e); } }
      return item.amount;
    };

    export const decreaseAmount = async function (itemId) {
      const item = cart.find(el => el.id == itemId);
      if (!item) return;
      if ((Number(item.amount) || 1) <= 1) return item.amount; // min 1, never auto-remove
      item.amount = (Number(item.amount) || 1) - 1;
      if (!localStorage.getItem('auth-token')) createLocalStorage();
      // Logged-in decrement has no dedicated server route; the local model.cart (which
      // drives the /cart page + checkout in this codebase) is updated and persisted on
      // the next mutation. Do NOT invent a backend route (out of scope: no new backend).
      return item.amount;
    };

<known_limitations>
## Disclosed limitation (conscious accept — surface in the SUMMARY)

**Logged-in decrement is not server-synced.** `/addtocart` lets us increment a logged-in user's server cart, but the backend has NO decrement route (and adding one is out of scope — "no new backend routes"). Therefore, for an AUTHENTICATED user, `decreaseAmount` updates `model.cart` + the UI in-session but does NOT lower the server-side count: on the next full reload the server still returns the higher count and the cart re-hydrates to it. In-session shared state (D-08 — drawer and /cart page agree while the page is loaded) still holds.

This is an accepted limitation, not a silent gap: it must be written into `43-02-SUMMARY.md` so the phase verification and any future "persist logged-in quantity edits" work is informed. The guest path (the dominant flow for this store, and the one that drives checkout via `model.cart` + localStorage) is fully persistent for both increase and decrease. If the `/addtocart` route check above forces the localStorage-authoritative branch, this asymmetry is reduced to "logged-in quantity edits are session-local on both directions" — still disclosed here.
</known_limitations>
