       ```js
       async function renderHomePage(req, res) {
         try {
           const pageData = buildPageData(req, 'home', []);
           const featuredProducts = await Product.find({
             isFeatured: true,
             available: { $ne: false },
             quantity: { $gt: 0 },
           })
             .sort({ featuredOrder: 1 })
             .limit(8)
             .select('id name slug images mainImage smallImages ils_price usd_price discount_percentage original_ils_price original_usd_price name_en name_he')
             .lean();
           pageData.featuredProducts = featuredProducts;
           res.render('pages/home', pageData);
         } catch (err) {
           console.error('Home SSR error:', err);
           const pageData = buildPageData(req, 'home', []);
           pageData.featuredProducts = [];
           res.render('pages/home', pageData);
         }
       }
       ```
    The filter (`available: { $ne: false }, quantity: { $gt: 0 }`) implements D-03 (skip out-of-stock); `.limit(8)` implements D-02 (cap 8); `.sort({ featuredOrder: 1 })` implements D-01 ordering. The `.select()` MUST include `id` (real numeric field) so `data-id` is non-empty under `.lean()`. On error, fall back to an empty featured list (NOT a 500) so the rest of the homepage still renders.
