```js
// Source: analog of displayOrder pre-save hook — Product.js lines 122–143
ProductSchema.pre('save', async function(next) {
  // Only assign featuredOrder when isFeatured is true and featuredOrder is blank
  if (this.isFeatured && this.featuredOrder == null) {
    try {
      const Product = this.constructor;
      // Find highest featuredOrder across ALL featured products (global, not per-category)
      const maxProduct = await Product.findOne({ isFeatured: true })
        .sort({ featuredOrder: -1 })
        .select('featuredOrder')
        .lean();

      this.featuredOrder = maxProduct?.featuredOrder
        ? maxProduct.featuredOrder + 10
        : 10;
    } catch (err) {
      console.error('Error assigning featuredOrder:', err);
    }
  }
  // Clear featuredOrder when product is un-featured
  if (!this.isFeatured && this.featuredOrder != null) {
    this.featuredOrder = null;
  }
  next();
});
```

```js
async function renderHomePage(req, res) {
  try {
    const pageData = buildPageData(req, 'home', []);

    const featuredProducts = await Product.find({
      isFeatured: true,
      available: { $ne: false },
      quantity: { $gt: 0 },
    })
      .sort({ featuredOrder: 1 })
      .limit(8)
      .select('id name slug images mainImage smallImages ils_price usd_price discount_percentage original_ils_price original_usd_price name_en name_he')
      .lean();

    pageData.featuredProducts = featuredProducts;

    res.render('pages/home', pageData);
  } catch (err) {
    console.error('Home SSR error:', err);
    // Fallback: render without featured products rather than hard 500
    const pageData = buildPageData(req, 'home', []);
    pageData.featuredProducts = [];
    res.render('pages/home', pageData);
  }
}
```

**Error handling pattern:** Copy the `try/catch` from `ssrDynamic.js` lines 123–126 (`console.error` + fallback render), but for home: fall back to empty featured list rather than 500, so the rest of the homepage still renders.

```js
// Read currency from localStorage — set by View.js / Phase 40 currency wiring
function getCurrency() {
  var c = (localStorage.getItem('currency') || 'ils').toLowerCase();
  return (c === 'usd' || c === 'ils') ? c : 'ils';
}
function symbol(cur) { return (cur || getCurrency()) === 'usd' ? '$' : '₪'; }
function money(n, cur) { return symbol(cur) + n; }
```

```js
  cards.forEach(function(card) {
    var usdPrice  = parseInt(card.dataset.usdPrice,  10) || 0;
```

```js
function priceHTML(p, cur) {
  var onSale = p.original != null && p.original > p.price;

    '<span class="tk-price__now' + (onSale ? ' is-sale' : '') + '">' + money(p.price, cur) + '</span>' +
    (onSale ? '<span class="tk-price__was">' + money(p.original, cur) + '</span>' : '') +
```

## Shared Patterns

### Currency Event — `currency-changed`
**Source:** `frontend/js/View.js` lines 9–72
**Apply to:** `frontend/js/homepage.js`

The event name is `'currency-changed'` (string constant). Detail shape: `{ currency: 'usd' | 'ils' }`. Dispatched by `View.js` on `<select name="currency">` change. `homepage.js` subscribes via:
```js
window.addEventListener('currency-changed', function(e) { ... });
```
`localStorage.getItem('currency')` returns `'usd'` or `'ils'` (lowercase). Use `getSavedCurrency` equivalent from View.js line 18–20.

### data-* Dual-Price Card Convention
**Source:** `backend/views/pages/category.ejs` line 103
**Apply to:** `backend/views/pages/home.ejs` `#tk-prod-grid` cards

All monetary values stored as `Math.round(product.X_price || 0)` integers in data-* attributes. Client reads via `card.dataset.usdPrice` / `card.dataset.ilsPrice` (camelCase conversion by browser). Never store formatted strings — always raw integers.

### SSR Product Query Pipeline
**Source:** `backend/routes/ssrDynamic.js` lines 65–72
**Apply to:** `backend/routes/ssr.js` `renderHomePage`

Standard pipeline: `Product.find(filter).sort(...).select(fieldList).lean()`. Always include `.lean()` for SSR performance. Always include `available: { $ne: false }, quantity: { $gt: 0 }` in filter. Add `.limit(8)` for featured query.

### Homepage Cache Invalidation
**Source:** `backend/cache/invalidation.js` lines 9–25

Home page cache keys (`/en:en:USD`, `/en:en:ILS`, `/he:he:USD`, `/he:he:ILS`) are already cleared by `invalidateCategory(categorySlug)`. For changes to `isFeatured` / `featuredOrder` (which don't belong to a single category), call the home page keys directly:

```js
// Add to invalidation.js — new exported function
function invalidateHomePage() {
  const homeKeys = ['/en:en:USD', '/en:en:ILS', '/he:he:USD', '/he:he:ILS'];
  homeKeys.forEach(key => pageCache.del(key));
  console.log('Cache invalidated for home page');
}
```

Call `invalidateHomePage()` from the product save/update endpoint in `backend/index.js` whenever `isFeatured` is toggled. Existing `invalidateProduct` / `invalidateCategory` calls remain untouched.

### Admin Card/Field Markup
**Source:** `admin/BisliView.js` lines 4015–4159
**Apply to:** `admin/BisliView.js` featured card block

Structure is always:
```
div.card > div.card__header > h3.card__title
         > div.card__body  > div.field > div.label + input.input + div.help
```
Checkbox label uses `<label style="display:flex; align-items:center; gap:8px">` (no wrapper `.field-label` class). `.help` is always a bare `<div class="help">` sibling of the input.

### SSR + Client Dual-Render Safety
**Source:** CLAUDE.md "SSR + Client Dual-Render" section

`home.ejs` ↔ `homepage.js` are a dual-render pair. Any structural change to the `#tk-prod-grid` section in `home.ejs` must be mirrored in `homepage.js` and vice-versa. The new `renderProducts()` must only update `.tk-prod__price` children — it must NOT call `grid.innerHTML = ''` or rebuild card DOM (that would destroy SSR-rendered content). `homepage.js` hydrates; it does not overwrite.
